void communication::forWards(vector<vector<double>>& li, vector<vector<double>>& w, vector<vector<double>>& lo)
{
int wt = w.size();
int ht = w[0].size();
for (int y = 0; y < ht; y++)
{
lo[y][0] = 0;
for (int x = 0; x < wt; x++)
{
lo[y][0] = lo[y][0] + li[x][0] * w[x][y];
}
lo[y][0] = 1 / (1 + exp(-1 * lo[y][0]));
//lo[y][0] = pow(tan(lo[y][0]), -1);
}
}
